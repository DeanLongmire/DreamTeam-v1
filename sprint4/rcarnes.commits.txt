const create_session = (req, res, leagueJSON, callback) =>{
    req.session.user = {
        id: leagueJSON.id,
        name: leagueJSON.name,
        sport: leagueJSON.sport
    }
    res.cookie('myCookie' + leagueJSON.id.substring(0,5), req.session.id);
    callback();
}
